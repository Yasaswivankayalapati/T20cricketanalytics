import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

# Load the data
df = pd.read_csv('t20_matches.csv')

# Inspect the data
print(df.head())
print(df.info())

# Count the number of wins for each team
wins = df['winner'].value_counts()

# Plot the number of wins for each team
plt.figure(figsize=(10, 6))
sns.barplot(x=wins.index, y=wins.values)
plt.title('Number of Wins by Team')
plt.xlabel('Team')
plt.ylabel('Number of Wins')
plt.xticks(rotation=90)
plt.show()

# Calculate the average scores for each team
average_scores = df.groupby('team1')['score1'].mean()

# Plot the average scores
plt.figure(figsize=(10, 6))
sns.barplot(x=average_scores.index, y=average_scores.values)
plt.title('Average Scores by Team')
plt.xlabel('Team')
plt.ylabel('Average Score')
plt.xticks(rotation=90)
plt.show()
